struct SNode *FindMiddle(LinkList &HA) {
    struct node *h = HA.head;
    int length = 0;
    if (h = nullptr)
        return nullptr;
    while (h- != nullptr) {
        length++;
        h = h->next;
    }
    length /= 2;
    h = HA.head;
    while (length-- > 0) {
        h = h->next;
    }
    return h;
}

    struct SNode *i = HA.head->next;
    struct SNode *j = HB.head->next;

void inverList(SeqList &L) {
    int n = L.length;
    for (int i = 0; i < n / 2; i++) {
        int t = L.list[i];
        L.list[i] = L.list[n - i - 1];
        L.list[n - i - 1] = t;
    }
}
