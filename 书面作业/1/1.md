# 第一次作业
## 第一题 
1. 计算$\sum\limits _{k=1}^{n} k!$，复杂度$O(n^2)$
```cpp
int sum(int n) {
    int s = 0;
    for (int i = 1; i <= n; i++) {
        int p = 1;
        for (int j = 1; j <= i; j++)  //计算i!
            p *= j;
        s += p;
    }
    return s;
}
```
2. 计算$\sum\limits _{k=1}^{n} k!$，复杂度$O(n)$
```cpp
int fac(int n) {
    int p = 1, s = 0;
    for (int i = 1; i <= n; i++) {
        p *= i;   //利用上个循环计算i!
        s += p;
    }
    return s;
}
```

```cpp
void f(int n) {
    int s, i = 0;
    while (s <= n) {//s~i^2<n
        ++i;
        s += i;   //计算1到i的和
        printf("*");
    }
}
```
2. $复杂度O(n(logn)^2)$
```cpp
void f(int n) {
    int i, j, k, count = 0;
    for (i = n / 2; i <= n; i++)          // n/2次
        for (j = 1; j < n; j *= 2)        // logn次
            for (k = 1; k <= n; k *= 2)   // logn次
                count++;
}
```

struct SNode *FindMiddle(LinkList &HA) {
    struct node *h = HA.head;
    int length = 0;
    if (h = nullptr)
        return nullptr;
    while (h- != nullptr) {
        length++;
        h = h->next;
    }
    length /= 2;
    h = HA.head;
    while (length-- > 0) {
        h = h->next;
