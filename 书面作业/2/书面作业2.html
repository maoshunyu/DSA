void next(char *P, int N[]) {
    int m = strlen(P);
    N[0] = 0;
    int i = 1, j = 0;
    while (i < m) {
        if (P[i] == P[j]) {
            N[i] = j + 1;
            i++;
            j++;
        } else if (j > 0)
            j = N[j - 1];
        else {
            N[i++] = 0;
        }
    }
}

int BinaryTree::FindMax(TNODE *root) {
    int root_val, left, right, max = INT_MIN;
    if (root != NULL) {
        root_val = root.value;
        left = FindMax(root.left);
        right = FindMax(root.right);
        max = max(left, right, root_val);
    }
    return max
}

int BinaryTree::FindLevelwithMaxSum(BinaryTree *root) {
    TNODE *temp;
    int level = 0, maxLevel = 0;
    struct Queue Q;
    int currentSum = 0, maxSum = 0;
    if (!root)
        return 0;
    InitQueue(Q);
    EnQueue(Q, root);
    while (!Q.isEmpty()) {
        int size = Q.size();
        while (size-- > 0) {
            temp = DeQueue(Q);
            currentSum += temp.value;

            if (temp.left)
                EnQueue(Q, temp.left);
            if (temp.right)
                EnQueue(Q, temp.right);
        }
        if (maxSum < currentSum)
            maxSum = currentSum;
        currentSum = 0;
    }
    return maxSum;
}

Graph *ReverseTheDirectedGraph(Graph *G) {
    Graph *Gr;
    if (G == NULL) {
        Gr = NULL;
        return NULL;
    }
    Initialize(Gr, Vcnt(G), NULL);
    int n = Vcnt(G);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (G->edge(i, j))
                Gr->insert(j, i);
        }
    }
}
